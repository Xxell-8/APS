```python
def bfs(graph, start):
    visited = [0] * (n+1)
    queue = [start]
    
    while queue:
        current = queue.pop(0)
        if not visited[current]:
            visited[current] = 1
            
            for edge in graph[current]:
                if not visited[edge]:
                    queue.append(edge)
```
