```python
def binary_search(arr, key):
    start = 0
    end = len(arr)-1
    
    while start <= end:
        middle = (start+end) // 2
        if arr[middle] == key:
            return True
        elif arr[middle] > key:
            end = middle -1
        else:
            start = middle + 1
    return False
```

```python
def binary_search(arr, low, high, key):

        return False
```
